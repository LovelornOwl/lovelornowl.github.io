~~~ cpp
#include "stdafx.h"
#include <vector>
#include <string>
#include <iostream>
using namespace std;

int main()
{
  int list0 []= { 0,2,4,6,8,10};
  int list1 [] = { 1,3,5,7,9};
  int list2 [size(list0)+size(list1)];
  int i = 0, j = 0, k = 0;
  while (i < size(list0) && j < size(list1)) {
    if (list0[i]<=list1[j]) {
      list2[k++] = list0[i++];
    }
    else {
      list2[k++] = list1[j++];
    }
  }
  return 0;
}
~~~
